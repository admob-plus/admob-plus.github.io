Rewarded Ad

Usage​

import React from 'react'
import AdMob, { RewardedAd } from '@admob-plus/react-native'

React.useEffect(async () => {
  await AdMobPlus.start()

  const rewarded = new RewardedAd({
    adUnitId: 'ca-app-pub-3940256099942544/5224354917',
  })
  await rewarded.load()
  await rewarded.show()
}, [])
